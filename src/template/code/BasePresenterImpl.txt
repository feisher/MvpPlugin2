public class BasePresenterImpl<V extends BaseView> implements BasePresenter<V>{
    protected V mView;
    Context mContext;
    Reference<V> viewRef;
    @Override
    public void attachView(V view) {
        viewRef = WeakReference(view)
         mView = viewRef!!.get() as V
         mContext = view.context()!!
    }

    @Override
    public void detachView() {
       if (viewRef != null) {
         viewRef!!.clear()
         viewRef = null
        }
    }
}
